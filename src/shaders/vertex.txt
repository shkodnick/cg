#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in int colorIndex;

out vec3 fragmentColor;

const vec3 colors[4] = vec3[](
    vec3(1.0, 0.0, 0.0),
    vec3(0.0, 1.0, 0.0),
    vec3(1.0, 0.0, 1.0),
    vec3(0.5, 0.5, 1.0)
);

void main()
{
    gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0);
    fragmentColor = colors[colorIndex];
}
